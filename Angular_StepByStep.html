@Input & @Output — Parent ↔ Child Communication
Components talk to each other — data pass pannuvom
Oru big component break panni small components create pannuvom. StudentList (parent) la StudentCard (child) components create pannuvom. Parent → Child data pass = @Input. Child → Parent event send = @Output.

// student-card.component.ts
import { Component, Input, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'app-student-card',
  standalone: true,
  template: `
    <div style="border:1px solid #ccc; padding:12px; margin:8px">
      <h3>{{ name }}</h3>           <!-- @Input value display -->
      <p>{{ course }}</p>
      <button (click)="onDelete()">Delete</button>
    </div>
  `
})
export class StudentCardComponent {

  // @Input — Parent irunthu data receive pannuvom
  @Input() name = '';
  @Input() course = '';
  @Input() studentId = 0;

  // @Output — Parent ku event send pannuvom
  @Output() deleted = new EventEmitter<number>();

  onDelete() {
    this.deleted.emit(this.studentId);  // Parent ku id send pannuvom
  }
}

// student.service.ts — Static data kooda (API illama)
import { Injectable } from '@angular/core';

@Injectable({
  providedIn: 'root'   // App everywhere use panlam — Singleton
})
export class StudentService {

  // Private data — component directly access panna vendaam
  private students = [
    { id: 1, name: 'Gurumoorthy', course: 'CS', age: 25 },
    { id: 2, name: 'Basha',       course: 'EE', age: 24 },
    { id: 3, name: 'Sridhar',     course: 'ME', age: 26 },
  ];

  // Methods — CRUD operations
  getAll() {
    return [...this.students];  // Copy return pannuvom — original change aagaathu
  }

  getById(id: number) {
    return this.students.find(s => s.id === id);
  }

  add(student: { name: string; course: string; age: number }) {
    const newId = Math.max(...this.students.map(s => s.id)) + 1;
    this.students.push({ id: newId, ...student });
  }

  delete(id: number) {
    this.students = this.students.filter(s => s.id !== id);
  }
}

// student.service.ts — HttpClient use panna update pannuvom
import { Injectable, inject } from '@angular/core';
import { HttpClient } from '@angular/common/http';      // ← Import
import { Observable } from 'rxjs';                        // ← Import
import { Student, ApiResponse } from './models/student.model';

@Injectable({ providedIn: 'root' })
export class StudentService {

  // API base URL — our .NET project port
  private apiUrl = 'http://localhost:5260/api/v2/students';

  // HttpClient inject pannuvom
  private http = inject(HttpClient);

  // ── GET ALL ────────────────────────────────────────────────
  // Observable<T> = async response — subscribe panna data varum
  getAll(): Observable<ApiResponse<Student[]>> {
    return this.http.get<ApiResponse<Student[]>>(this.apiUrl);
  }

  // ── GET BY ID ──────────────────────────────────────────────
  getById(id: number): Observable<ApiResponse<Student>> {
    return this.http.get<ApiResponse<Student>>(`${this.apiUrl}/${id}`);
  }

  // ── CREATE (POST) ──────────────────────────────────────────
  create(student: Omit<Student, 'id'>): Observable<ApiResponse<Student>> {
    return this.http.post<ApiResponse<Student>>(this.apiUrl, student);
  }

  // ── UPDATE (PUT) ───────────────────────────────────────────
  update(id: number, student: Student): Observable<ApiResponse<Student>> {
    return this.http.put<ApiResponse<Student>>(`${this.apiUrl}/${id}`, student);
  }

  // ── DELETE ─────────────────────────────────────────────────
  delete(id: number): Observable<ApiResponse<any>> {
    return this.http.delete<ApiResponse<any>>(`${this.apiUrl}/${id}`);
  }
}
http.get() = .NET la HttpClient.GetAsync() mathiri. Observable return pannuvom — .NET la Task mathiri. Type parameter <ApiResponse<Student[]>> = "API idha maathiri response tharum" nu TypeScript ku solrom. URL = backtick `` use pannuvom template string ku — ${} la variable use pannalam.

students.component.ts — API call with loading + error
import { Component, OnInit, inject } from '@angular/core';
import { StudentService } from '../student.service';
import { Student } from '../models/student.model';
import { CommonModule } from '@angular/common';   // @for, @if ku

@Component({
  selector: 'app-students',
  standalone: true,
  imports: [CommonModule],
  templateUrl: './students.component.html'
})
export class StudentsComponent implements OnInit {

  private studentService = inject(StudentService);

  // State variables
  students: Student[] = [];
  isLoading = false;      // API call pandra pothu true
  errorMessage = '';      // Error varum pothu message store

  ngOnInit() {
    this.loadStudents();   // Component ready aana load
  }

  loadStudents() {
    this.isLoading = true;
    this.errorMessage = '';

    this.studentService.getAll().subscribe({

      next: (response) => {                      // ✅ API success
        if (response.success) {
          this.students = response.data;         // Data store pannuvom
        }
        this.isLoading = false;
      },

      error: (err) => {                          // ❌ API failed
        this.errorMessage = 'Failed to load students: ' + err.message;
        this.isLoading = false;
      }

    });
  }

  deleteStudent(id: number) {
    if (!confirm('Delete pannalama?')) return;

    this.studentService.delete(id).subscribe({
      next: () => this.loadStudents(),   // Success → refresh list
      error: (err) => alert('Delete failed!')
    });
  }
}

// add-student.component.ts
import { Component, Output, EventEmitter, inject } from '@angular/core';
import { FormBuilder, ReactiveFormsModule, Validators } from '@angular/forms';
import { StudentService } from '../student.service';

@Component({
  selector: 'app-add-student',
  standalone: true,
  imports: [ReactiveFormsModule],   // ← Required for reactive forms
  templateUrl: './add-student.component.html'
})
export class AddStudentComponent {

  private fb = inject(FormBuilder);
  private studentService = inject(StudentService);

  @Output() studentAdded = new EventEmitter<void>();  // Parent notify

  isSubmitting = false;
  successMessage = '';
  errorMessage = '';

  // Form define pannuvom with validations
  studentForm = this.fb.group({
    name:   ['', [Validators.required, Validators.minLength(3)]],
    email:  ['', [Validators.required, Validators.email]],
    course: ['', Validators.required],
    age:    [null, [Validators.required, Validators.min(10), Validators.max(99)]]
  });

  // Shortcut — form controls easy access ku
  get f() { return this.studentForm.controls; }

  onSubmit() {
    if (this.studentForm.invalid) return;  // Validation fail = stop

    this.isSubmitting = true;
    const formData = this.studentForm.value as any;

    this.studentService.create(formData).subscribe({
      next: (res) => {
        this.successMessage = 'Student added successfully!';
        this.studentForm.reset();             // Form clear pannuvom
        this.studentAdded.emit();             // Parent ku notify
        this.isSubmitting = false;
      },
      error: (err) => {
        this.errorMessage = 'Failed to add student!';
        this.isSubmitting = false;
      }
    });
  }
}
